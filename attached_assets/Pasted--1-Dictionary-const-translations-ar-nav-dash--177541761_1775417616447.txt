let currentLang = 'ar';

// 2. دالة تبديل اللغة الشاملة
function toggleLanguage() {
    currentLang = currentLang === 'ar' ? 'en' : 'ar';
    
    // تغيير فئة الجسم للتحكم في الاتجاه (RTL/LTR) والخطوط
    document.body.className = currentLang === 'ar' ? 'ar-mode' : 'en-mode';
    document.body.dir = currentLang === 'ar' ? 'rtl' : 'ltr';

    // تحديث كل النصوص التي تحمل خاصية data-i18n
    document.querySelectorAll('[data-i18n]').forEach(el => {
        const key = el.getAttribute('data-i18n');
        if (translations[currentLang][key]) {
            el.innerText = translations[currentLang][key];
        }
    });

    // تحديث نصوص الجداول والمكتبة
    updateLibraryUI();
    document.getElementById('lang-btn-text').innerText = translations[currentLang].lang_label;
}

// 3. قاعدة البيانات الهندسية (Expert Database)
const expertData = [
    { id: "MAF", sys: "Engine", volt: "0.5-4.5V", logic_ar: "قياس كتلة الهواء الداخلة", logic_en: "Measuring intake air mass" },

    { id: "CKP", sys: "Ignition", volt: "AC Sine", logic_ar: "تحديد توقيت الحقن والشرارة", logic_en: "Crankshaft position & RPM" }
];

// 4. تحديث واجهة المكتبة بناءً على اللغة
function updateLibraryUI() {
    const tbody = document.getElementById('library-body');
    if (!tbody) return;
    
    tbody.innerHTML = '';
    expertData.forEach(item => {
        tbody.innerHTML += `
            <tr style="border-bottom: 1px solid #333;">
                <td style="padding: 12px; color: var(--neon-blue); font-weight: bold;">${item.id}</td>
                <td style="padding: 12px;">${item.sys}</td>
                <td style="padding: 12px; color: orange;">${item.volt}</td>
                <td style="padding: 12px;">${currentLang === 'ar' ? item.logic_ar : item.logic_en}</td>
            </tr>
        `;
    });
}

// 5. دالة الفحص الذكي (AI Diagnostic)
function runDiagnostic() {
    const log = document.getElementById('scan-log');
    log.innerHTML = `<p class="blink">${translations[currentLang].scanning_msg}</p>`;
    
    setTimeout(() => {
        log.innerHTML = `
            <div style="border: 1px solid var(--launch-red); padding: 15px; color: var(--launch-red); background: rgba(230,0,18,0.1);">
                <i class="fas fa-exclamation-triangle"></i> ${translations[currentLang].fault_msg}
            </div>
        `;
        // إذا كان هناك مجسم 3D، نغير لونه هنا
        if(window.carMesh) window.carMesh.material.color.setHex(0xe60012);
    }, 2500);
}

// تهيئة النظام عند التحميل
window.onload = () => {
    updateLibraryUI();
    // تشغيل الساعة
    setInterval(() => {
        document.getElementById('digital-clock').innerText = new Date().toLocaleTimeString();
    }, 1000);
};
